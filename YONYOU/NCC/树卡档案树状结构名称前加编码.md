package nccloud.web.insbd.busisource.bd_busisource.action;

import nccloud.commons.lang.ArrayUtils;
import nccloud.web.codeplatform.framework.action.base.BaseAction;
import nccloud.web.codeplatform.framework.action.base.RequestDTO;
import nccloud.web.codeplatform.framework.action.base.RequstParamWapper;
import nccloud.web.codeplatform.framework.action.base.VOTransform;
import nccloud.web.insbd.commons.tree.TreeBuilderPid;
import nccloud.framework.service.ServiceLocator;
import nccloud.framework.web.container.IRequest;
import nc.vo.insbd.busisource.BusiSourceVO;

import nc.itf.insbd.busisource.bd_busisource.IBusiSourceVOService;

public class LoadTreeBusiSourceVOAction extends BaseAction{

	@Override
	public Object doAction(IRequest request, RequstParamWapper paramWapper) throws Throwable {
		//json数据转换
		RequestDTO param = VOTransform.toVO(paramWapper.requestString,RequestDTO.class);
		
		IBusiSourceVOService service = ServiceLocator.find(IBusiSourceVOService.class);
		BusiSourceVO[] vos = service.loadTreeData(BusiSourceVO.class,param.getUserJson());
		
		if(ArrayUtils.isEmpty(vos)) {
			return null;
		}

		return buildResult(new TreeBuilderPid(),vos);	// Edited by Jiahui Li 20260227	
	}

}

package nccloud.web.insbd.commons.tree;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

import nccloud.web.codeplatform.framework.commons.tree.ITreeBuilder;
import nccloud.web.codeplatform.framework.commons.tree.ITreeNode;
import nccloud.web.codeplatform.framework.commons.tree.RootTreeNode;

/**
 * 树状结构名称前加编码
 * 
 * @author Jiahui Li
 * @date 2026-02-27
 *
 */
public class TreeBuilderPid implements ITreeBuilder {
	RootTreeNode root = new RootTreeNode();

	@Override
	public List<ITreeNode> buildTree(List<ITreeNode> nodes) {
		if (nodes != null && !nodes.isEmpty()) {

			Map<String, ITreeNode> nodeMap = (Map) nodes.stream()
					.collect(Collectors.toMap(ITreeNode::getRefpk, v -> v, (k1, k2) -> k2));
			nodes.stream().forEach(node -> {
				ITreeNode pNode = (ITreeNode) nodeMap.get(node.getPid());

				// 名称前加编码
				if (node != null) {
					String code = node.getCode();
					String name = node.getTitle();

					node.setTitle(code + " " + name);
					node.setRefname(code + " " + name);
				}

				if (pNode != null) {
					pNode.getChildren().add(node);
				} else {
					this.root.getChildren().add(node);
				}
			});
			List<ITreeNode> list = new ArrayList<ITreeNode>();

			list.addAll(this.root.getChildren());
			return list;
		}
		return null;
	}

}
